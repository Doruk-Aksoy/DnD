{
  Inventory.PickupMessage "You got a field kit"
  Inventory.PickupSound "Misc/I_PkUp"

  Inventory.MaxAmount 3
  Inventory.InterHubAmount 3

		TNT1 A 0 A_GiveInventory("DnD_QuestReward_ArtiDmgPower", 1)
	Stop
	Failed:
		TNT1 A 1
	Fail
  }
}
