// args4 for pellet count, args3 for damage, arg2 for starting pellet count
Actor Pellets : DnD_Activator {
	States {
		Use:
			TNT1 A 0 A_SetArg(4, 20)
		LoopStart:
			TNT1 A 0 A_JumpIf(Args[4], "LoopContinue")
		Goto Finish
		LoopContinue:
			TNT1 A 0 A_FireBullets((360.0 / 20) * Args[4], 0, -1, 15, "NewBulletPuff", FBF_EXPLICITANGLE)
			TNT1 A 0 A_SetArg(4, Args[4] - 1)
		Goto LoopStart
		Finish:
			TNT1 A 0
		Stop
	}
}
