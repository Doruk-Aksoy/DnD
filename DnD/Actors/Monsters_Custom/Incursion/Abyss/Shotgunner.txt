ACTOR AbyssalShotgunner : DnD_BaseRangedMonster {
	Health 0x7FFFFFFF
	Radius 20
	Height 56
	Mass 100
	Speed 8
	PainChance 155
	Monster
	DropItem "ShotgunPickup_Drop"
	SeeSound "abyssal_human/sight"
	AttackSound "abyssal_human/shotgun"
	PainSound "abyssal_human/pain"
	DeathSound "abyssal_human/death"
	ActiveSound "abyssal_human/active"
	Obituary "%o was hunted down by an Abyssal Shotgunner."
	Tag "Abyssal Shotgunner"
	Species "Shotgunner"
	Decal BulletChip
	Accuracy 30
	Scale 0.75
	+ISMONSTER
	States {
		SetInfo:
			TNT1 A 0 A_SetArg(4, DND_ABYSS_SHOTGUNNER)
		Goto SpawnSetup
	
		IdleState:
			C015 AB 10 A_Look
		Loop
		SeeState:
			C015 AABBCCDD 4 A_Chase
		Loop
		MissileState:
			C015 E 10 A_FaceTarget
			TNT1 A 0 A_PlaySound("abyssal_human/shotgun", CHAN_WEAPON)
			TNT1 AAA 0 A_CustomMissile("ZombieBullet_Abyssal", 32, 0, frandom(-22.5, 22.5))
			C015 F 10 Bright A_CustomMissile("ZombieBullet_Abyssal", 32, 0, frandom(-22.5, 22.5))
			C015 E 10 A_SpawnItemEx ("shotguncasing", -8, 7, 26, 4, 0, frandom(3.0, 5.0), random(-80, -90))
		Goto See
		PainState:
			C015 G 3
			C015 G 3 A_Pain
		Goto See
		DeathState:
			C015 H 5
			C015 I 5 A_Scream
			C015 J 5 A_NoBlocking
			C015 K 5
			C015 L -1
		Stop
		XDeathState:
			TNT1 A 0 A_SpawnDebris("NashGore_Gib9",0)
			TNT1 A 0 A_SpawnDebris("NashGore_Gib10",0)
			TNT1 A 0 A_SpawnDebris("NashGore_Gib11",0)
			TNT1 A 0 A_SpawnDebris("NashGore_Gib11",0)
			C015 M 5 A_SpawnItemEx("GenericGibber_Low", 0, 0, 0, 0, 0, 0, 0, SXF_USEBLOODCOLOR)
			C015 N 5 A_XScream
			C015 O 5 A_NoBlocking
			C015 PQRST 5
			C015 U -1
		Stop
		Raise:
			C015 KJIH 5
		Goto See	
	}
}

ACTOR AbyssalShotgunner_Elite : AbyssalShotgunner { 
	States {
		SetInfo:
			TNT1 A 0 A_SetArg(3, DND_MONSTERIS_ELITE)
			TNT1 A 0 A_SetArg(4, DND_ABYSS_SHOTGUNNER)
		Goto SpawnSetup
	}
}
